  <script type="module">
  document.addEventListener("DOMContentLoaded", () => {
    const btn = document.getElementById("connect-btn");
    let provider = window?.solana || window?.phantom?.solana;

    // --- Wallet Connect / Disconnect ---
    async function connectWallet() {
      if (!provider) return alert("No Solana wallet found ‚Äî install Phantom/Solflare/Backpack.");

      // If already connected ‚Üí disconnect
      if (provider.isConnected) {
        await provider.disconnect().catch(console.error);
        sessionStorage.removeItem("walletConnected");
        updateButtonState(null);
        return;
      }

      try {
        const { publicKey } = await provider.connect();
        const address = publicKey.toString();
        sessionStorage.setItem("walletConnected", "true");
        updateButtonState(address);
        console.log("Connected:", address);
      } catch (err) {
        console.log("Connection cancelled:", err);
      }
    }

    function updateButtonState(address) {
      if (!btn) return;
      if (address) {
        btn.innerText = "Connected";
        btn.className = "px-6 py-2 rounded-full text-xs font-semibold tracking-wide bg-white text-black border border-green-500 hover:scale-105 transition";
      } else {
        btn.innerText = "Connect Wallet";
        btn.className = "px-6 py-2 rounded-full text-xs font-semibold tracking-wide bg-green-500 text-black hover:bg-green-400 hover:scale-105 transition";
      }
    }

    // Restore previous session if wallet connected
    if (sessionStorage.getItem("walletConnected") === "true" && provider?.publicKey) {
      updateButtonState(provider.publicKey.toString());
    }

    // Listen to wallet events
    if (provider) {
      provider.on("connect", () => updateButtonState(provider.publicKey?.toString()));
      provider.on("disconnect", () => updateButtonState(null));
    }

    btn?.addEventListener("click", connectWallet);

    // --- Tab Navigation ---
    window.showView = function(view) {
      const views = ["home","voting","marketplace","greenpaper","game"];
      views.forEach(v=>{
        const el = document.getElementById(`view-${v}`);
        if(el) el.classList.add("hidden");
      });

      const activeView = document.getElementById(`view-${view}`);
      if(activeView) activeView.classList.remove("hidden");

      // Update nav link colors
      document.querySelectorAll(".nav-link").forEach(el=>{
        if(el) el.classList.remove("text-green-400");
      });
      const activeBtn = Array.from(document.querySelectorAll(".nav-link")).find(b => b.textContent.toLowerCase() === view);
      if(activeBtn) activeBtn.classList.add("text-green-400");
    }
  });
  </script>
